const = 1241245125.124214 # некое значение постоянное
print(const) # печать на экран полного значения
print('Константа:',const) # печать на экран заданного текста и полного значения
print('Константа:{}'.format(const)) # аналогично, но с помощью специальной обработки (пока без доп условий)
print('Константа:{:.2f}'.format(const)) # печать значения с отбрасыванием знаков после второго знака после запятой
print('Константа:{:.2%}'.format(const)) # печать значения в процентах, со знаком процента (значение, умноженное на 100)
